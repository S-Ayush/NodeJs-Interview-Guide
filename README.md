### Example: Event Loop Explanation (Chapter 2)

```javascript
// Understanding Event Loop Order
console.log('1');                              // Sync
setTimeout(() => console.log('2'), 0);         // Macrotask
Promise.resolve().then(() => console.log('3')); // Microtask
process.nextTick(() => console.log('4'));      // Highest priority
console.log('5');                              // Sync

// Output: 1, 5, 4, 3, 2
// Why? Sync â†’ nextTick â†’ Promises â†’ setTimeout
```

### Example: Express Middleware (Chapter 6)

```javascript
// Custom Authentication Middleware
const authMiddleware = (req, res, next) => {
  const token = req.headers.authorization?.split(' ')[1];

  if (!token) {
    return res.status(401).json({ error: 'No token provided' });
  }

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    req.user = decoded;
    next(); // MUST call next() to continue
  } catch (err) {
    res.status(401).json({ error: 'Invalid token' });
  }
};

// Usage
app.get('/protected', authMiddleware, (req, res) => {
  res.json({ user: req.user });
});
```

### Example: Interview Question (Chapter 12)

**Q: Explain the difference between process.nextTick() and setImmediate()**

```javascript
// Answer:
// process.nextTick() executes before the event loop continues
// setImmediate() executes in the check phase of event loop

// Example:
setImmediate(() => console.log('immediate'));
process.nextTick(() => console.log('nextTick'));
console.log('sync');

// Output: sync, nextTick, immediate

// EXCEPTION: Inside I/O cycle, setImmediate runs first
fs.readFile('file.txt', () => {
  setImmediate(() => console.log('immediate')); // First
  setTimeout(() => console.log('timeout'), 0);  // Second
});
```
